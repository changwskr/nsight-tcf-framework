NSIGHT 17 WAR 단일 WAS + DeltaManager 설정 패키지

구조:
nh.marketing.com -> Apache -> Tomcat Cluster(DeltaManager) -> 17 WAR

업무 WAR:
cc, ic, pc, bc, ms, sv, pd, cm, eb, ep, bp, bd, ss, cs, ct, mg

[apache/httpd.conf]
ServerRoot "/app/apache"
Listen 80
Listen 443
ServerName nh.marketing.com
ServerTokens Prod
ServerSignature Off
TraceEnable Off
Timeout 30
KeepAlive On
KeepAliveTimeout 5
MaxKeepAliveRequests 100
LoadModule proxy_module modules/mod_proxy.so
LoadModule proxy_http_module modules/mod_proxy_http.so
LoadModule proxy_balancer_module modules/mod_proxy_balancer.so
LoadModule slotmem_shm_module modules/mod_slotmem_shm.so
LoadModule lbmethod_byrequests_module modules/mod_lbmethod_byrequests.so
LoadModule ssl_module modules/mod_ssl.so
LoadModule rewrite_module modules/mod_rewrite.so
LoadModule headers_module modules/mod_headers.so
Include conf/extra/nsight-balancer.conf
Include conf/extra/nsight-vhost.conf

[apache/extra/nsight-balancer.conf]
<Proxy "balancer://nsight-marketing-was-cluster">
  BalancerMember "http://10.10.10.11:8080" route=was1 connectiontimeout=3 timeout=10 retry=0
  BalancerMember "http://10.10.10.12:8080" route=was2 connectiontimeout=3 timeout=10 retry=0
  ProxySet lbmethod=byrequests stickysession=JSESSIONID|jsessionid scolonpathdelim=On
</Proxy>

[apache/extra/nsight-vhost.conf]
<VirtualHost *:80>
  ServerName nh.marketing.com
  RewriteEngine On
  RewriteRule ^/(.*)$ https://nh.marketing.com/$1 [R=301,L]
</VirtualHost>

<VirtualHost *:443>
  ServerName nh.marketing.com

[tomcat/conf/server.xml]
<Server port="8005" shutdown="SHUTDOWN">
  <Service name="Catalina">
    <Connector port="8080" protocol="HTTP/1.1"
      maxThreads="500" minSpareThreads="100" acceptCount="500"
      maxConnections="10000" connectionTimeout="8000"
      keepAliveTimeout="5000" maxKeepAliveRequests="100"
      URIEncoding="UTF-8" />
    <Engine name="Catalina" defaultHost="localhost" jvmRoute="was1">
      <Host name="localhost" appBase="webapps" unpackWARs="true" autoDeploy="false">
        <Cluster className="org.apache.catalina.ha.tcp.SimpleTcpCluster">
          <Manager className="org.apache.catalina.ha.session.DeltaManager"
            expireSessionsOnShutdown="false" notifyListenersOnReplication="true" />
        </Cluster>
      </Host>
    </Engine>
  </Service>
</Server>
