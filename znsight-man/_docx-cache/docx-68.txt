72. 목록조회 + 페이징 샘플
72.1 도입 전 안내말
본 장은 NSIGHT TCF Framework에서 목록조회와 페이징을 업무팀이 자체적으로 구현하는 표준 샘플을 정의한다.
페이징은 TCF Core가 관리하지 않는다.TCF는 기존처럼 전문 수신, Header 검증, 세션, 권한, 거래통제, Timeout, 거래로그, 표준응답 조립만 담당한다. 실제 페이징 처리는 각 업무 WAR 내부에서 Rule → Service → DAO → MyBatis 흐름으로 처리한다. 기존 페이징 설계 기준에서도 TCF Core·Dispatcher는 페이징에 관여하지 않고, 업무 WAR 내부에서 처리하도록 정리되어 있다.
이번 샘플은 SV 고객목록조회 거래로 정의한다.

package com.nh.nsight.marketing.sv.dto;

import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

@Getter
@Setter
@NoArgsConstructor
public class PageResponse {

    private int pageNo;
    private int pageSize;
    private int totalCount;
    private int totalPage;
    private boolean hasNext;

    public PageResponse(int pageNo, int pageSize, int totalCount) {
        this.pageNo = pageNo;
        this.pageSize = pageSize;
        this.totalCount = totalCount;
        this.totalPage = (int) Math.ceil((double) totalCount / pageSize);
        this.hasNext = pageNo < this.totalPage;
    }
}

72.7 Handler 샘플
package com.nh.nsight.marketing.sv.entry.handler;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.nh.nsight.marketing.sv.dto.SvCustomerListRequest;
import com.nh.nsight.marketing.sv.entry.facade.SvCustomerFacade;
import com.nh.nsight.tcf.core.context.TransactionContext;
import com.nh.nsight.tcf.core.handler.TransactionHandler;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

@Component("SV.Customer.selectList")
@RequiredArgsConstructor
public class SvCustomerListHandler implements TransactionHandler {

    private final SvCustomerFacade svCustomerFacade;
    private final ObjectMapper objectMapper;

    @Override
    public String getServiceId() {
        return "SV.Customer.selectList";
    }

    @Override
    public Object handle(TransactionContext context, Object body) {
        SvCustomerListRequest request =
                objectMapper.convertValue(body, SvCustomerListRequest.class);

        return svCustomerFacade.selectCustomerList(context, request);
    }
}

Handler는 업무 로직을 직접 처리하지 않는다.요청 Body를 DTO로 변환한 뒤 Facade를 호출한다. TCF 구조에서는 header.serviceId를 기준으로 Dispatcher가 Handler를 선택한다.

72.9 Service 샘플
package com.nh.nsight.marketing.sv.application.service;

import com.nh.nsight.marketing.sv.application.rule.SvCustomerRule;
import com.nh.nsight.marketing.sv.dto.PageResponse;
import com.nh.nsight.marketing.sv.dto.SvCustomerListItem;
import com.nh.nsight.marketing.sv.dto.SvCustomerListRequest;
import com.nh.nsight.marketing.sv.dto.SvCustomerListResponse;
import com.nh.nsight.marketing.sv.persistence.dao.SvCustomerDao;
import com.nh.nsight.tcf.core.context.TransactionContext;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

@Service
@RequiredArgsConstructor
public class SvCustomerService {

    private final SvCustomerRule svCustomerRule;
    private final SvCustomerDao svCustomerDao;

    public SvCustomerListResponse selectCustomerList(
            TransactionContext context,
            SvCustomerListRequest request) {

        svCustomerRule.validateCustomerListRequest(request);
        svCustomerRule.applyDefaultPage(request);
        svCustomerRule.calculateOffset(request);

        int totalCount = svCustomerDao.countCustomerList(request);

        List<SvCustomerListItem> list = List.of();
        if (totalCount > 0) {
            list = svCustomerDao.selectCustomerList(request);
        }

        PageResponse page = new PageResponse(
                request.getPage().getPageNo(),
                request.getPage().getPageSize(),
                totalCount
        );

        return new SvCustomerListResponse(list, page);
    }
}

72.10 Rule 샘플
package com.nh.nsight.marketing.sv.application.rule;

import com.nh.nsight.marketing.sv.dto.PageRequest;
import com.nh.nsight.marketing.sv.dto.SvCustomerListRequest;
import com.nh.nsight.tcf.core.exception.BusinessException;
import org.springframework.stereotype.Component;

@Component
public class SvCustomerRule {

    private static final int DEFAULT_PAGE_NO = 1;
    private static final int DEFAULT_PAGE_SIZE = 100;
    private static final int MAX_PAGE_SIZE = 500;

    public void validateCustomerListRequest(SvCustomerListRequest request) {
        if (request == null) {
            throw new BusinessException("E-SV-VAL-0101", "요청 정보가 없습니다.");
        }

        if (request.getSearchCondition() == null) {
            throw new BusinessException("E-SV-VAL-0102", "검색조건이 없습니다.");
        }

        if (request.getSearchCondition().getFromDate() != null
                && !request.getSearchCondition().getFromDate().matches("\\d{8}")) {
            throw new BusinessException("E-SV-VAL-0103", "조회 시작일자 형식이 올바르지 않습니다.");
        }

        if (request.getSearchCondition().getToDate() != null
                && !request.getSearchCondition().getToDate().matches("\\d{8}")) {
            throw new BusinessException("E-SV-VAL-0104", "조회 종료일자 형식이 올바르지 않습니다.");
        }
    }

    public void applyDefaultPage(SvCustomerListRequest request) {
        if (request.getPage() == null) {
            request.setPage(new PageRequest());
        }

        PageRequest page = request.getPage();

        if (page.getPageNo() == null || page.getPageNo() < 1) {
            page.setPageNo(DEFAULT_PAGE_NO);
        }

        if (page.getPageSize() == null || page.getPageSize() < 1) {
            page.setPageSize(DEFAULT_PAGE_SIZE);
        }

        if (page.getPageSize() > MAX_PAGE_SIZE) {
            throw new BusinessException(
                    "E-SV-VAL-0105",
                    "한 번에 조회할 수 있는 최대 건수를 초과했습니다."
            );
        }
    }

    public void calculateOffset(SvCustomerListRequest request) {
        PageRequest page = request.getPage();

        int offset = (page.getPageNo() - 1) * page.getPageSize();

        page.setOffset(offset);
        page.setLimit(page.getPageSize());
    }
}

페이징 계산 기준은 다음과 같다.
항목
계산식
pageNo
화면 요청값, 없으면 1
pageSize
화면 요청값, 없으면 100
offset
(pageNo - 1) × pageSize
limit
pageSize
totalPage
ceil(totalCount / pageSize)
예를 들어 pageNo=2, pageSize=10이면 offset = (2 - 1) × 10 = 10이다. 즉, DB는 앞의 10건을 건너뛰고 다음 10건을 조회한다.

72.14 Mapper XML 샘플 — Oracle 11g 이하 ROWNUM 방식
Oracle 11g 이하 또는 OFFSET FETCH를 사용할 수 없는 경우에는 ROWNUM 방식으로 작성한다.
<select id="selectCustomerListByRownum"
        parameterType="com.nh.nsight.marketing.sv.dto.SvCustomerListRequest"
        resultType="com.nh.nsight.marketing.sv.dto.SvCustomerListItem">

    SELECT *
    FROM (
        SELECT
              INNER_QUERY.*
            , ROWNUM AS RN
        FROM (
            SELECT
                  C.CUSTOMER_NO           AS customerNo
                , C.CUSTOMER_NAME         AS customerName
                , C.CUSTOMER_GRADE        AS customerGrade
                , C.BRANCH_CODE           AS branchCode
                , C.BRANCH_NAME           AS branchName
                , C.TOTAL_BALANCE         AS totalBalance
                , C.PRODUCT_COUNT         AS productCount
                , C.LAST_TRANSACTION_DATE AS lastTransactionDate
            FROM RDW_CUSTOMER_SUMMARY C
            WHERE 1 = 1

            <if test="searchCondition.customerName != null and searchCondition.customerName != ''">
                AND C.CUSTOMER_NAME LIKE '%' || #{searchCondition.customerName} || '%'
            </if>

            <if test="searchCondition.customerGrade != null and searchCondition.customerGrade != ''">
                AND C.CUSTOMER_GRADE = #{searchCondition.customerGrade}
            </if>

            ORDER BY C.LAST_TRANSACTION_DATE DESC, C.CUSTOMER_NO ASC
        ) INNER_QUERY
        WHERE ROWNUM <![CDATA[<=]]> #{page.offset} + #{page.limit}
    )
    WHERE RN <![CDATA[>]]> #{page.offset}

</select>

운영 표준에서는 가능하면 Oracle 12c 이상 기준의 OFFSET FETCH 방식을 우선 사용하고, 구버전 호환이 필요한 경우에만 ROWNUM 방식을 사용한다.

72.21 마무리말
목록조회 + 페이징 샘플의 핵심은 TCF는 페이징을 모른다는 점이다.
TCF
= 전문 수신
+ Header 검증
+ 세션 / 권한
+ 거래통제
+ Timeout
+ 거래로그
+ 표준응답

업무 WAR
= 검색조건 검증
+ pageNo / pageSize 검증
+ offset 계산
+ count SQL
+ list SQL
+ list + page 응답 조립

따라서 신규 목록조회 거래는 본 샘플에서 다음 항목만 업무에 맞게 바꾸면 된다.
변경 대상
예시
ServiceId
SV.Customer.selectList → CM.Campaign.selectList
DTO
고객 검색조건 → 캠페인 검색조건
Mapper SQL
RDW_CUSTOMER_SUMMARY → 업무 테이블
오류코드
E-SV-* → E-CM-*
거래코드
SV-INQ-0002 → CM-INQ-0001

72.22 시사점
목록조회는 단건조회보다 장애 위험이 높다.조건 없는 대량조회, Page Size 무제한, Count SQL 과다 수행, 정렬 기준 누락은 운영 장애로 이어질 수 있다.
NSIGHT 목록조회 페이징의 기준은 “업무팀 자체 처리, DB 레벨 페이징, Page Size 제한, Count/List SQL 분리, 거래로그 기록”이다.
